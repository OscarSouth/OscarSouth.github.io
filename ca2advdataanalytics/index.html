%%R
# calculate row means and convert into vector
means.c <- data.c %>%
  as_tibble() %>%
  transmute(mean=rowMeans(.)) %>%
  unlist() %>%
  as.vector()

# converting sample means into pHat for simulated data
pHat.c <- means.c/n

# calculating variences 
var1 <- n * pHat.c[1] * (1 - pHat.c[1])
var2 <- n * pHat.c[2] * (1 - pHat.c[2])
var3 <- n * pHat.c[3] * (1 - pHat.c[3])

# calculating covariences
cov12 <- -n * pHat.c[1] * pHat.c[2]
cov13 <- -n * pHat.c[1] * pHat.c[3]
cov23 <- -n * pHat.c[2] * pHat.c[3]

# calculating 
corr12 <- cov12 / sqrt(var1 * var2)
corr13 <- cov13 / sqrt(var1 * var3)
corr23 <- cov23 / sqrt(var2 * var3)
vec_corr <- c(corr12, corr13, corr23)

vec_corr[3] # -0.4839135 for present simulation, 
            # a moderately negative correllation

%%R -w 5 -h 8 --units in -r 200
# parameters & statistics
alpha.gam <- 3
lambda.gam <- 1 # lambda assigned as 1 as no lambda specified
n.gam <- seq(from=1, to=100, by=1)
beta.gam <- sqrt(1/lambda.gam)
mu.gam <- alpha.gam*beta.gam
sigma.gam <- sqrt(alpha.gam/(1/beta.gam)^2) # alt <- sqrt(alpha.gam*beta.gam^2)

# generate data
range.gam <- seq(from=1, to=100, by=1)
y1.gam <- rgamma(n=length(range.gam), shape=alpha.gam, rate=lambda.gam)
y2.gam <- dgamma(x=range.gam, shape=alpha.gam, rate=lambda.gam)
y3.gam <- pgamma(q=range.gam, shape=alpha.gam, rate=lambda.gam)

# plot
rplt.gam <- ggplot(tibble(range.gam, y1.gam), aes(x=y1.gam, y=range.gam)) +
  geom_point(size=0.1) + 
  labs(title='Gamma Distribution, alpha=3, lambda=1', x=NULL, y='value') +
  geom_vline(xintercept=mu.gam, size=0.2) +
  xlim(0, 15) +
  annotate('rect', xmin=mu.gam-sigma.gam, xmax=mu.gam+sigma.gam,
           ymin=-Inf, ymax=Inf, alpha=0.1) 
dplt.gam <- ggplot(tibble(range.gam, y2.gam), aes(x=range.gam, y=y2.gam)) +
  geom_line() +
  geom_vline(xintercept=mu.gam, size=0.2) +
  labs(title='', x=NULL, y='f(x)') +
  xlim(0, 15) +
  annotate('rect', xmin=mu.gam-sigma.gam, xmax=mu.gam+sigma.gam,
           ymin=-Inf, ymax=Inf, alpha=0.1) 
pplt.gam <- ggplot(tibble(range.gam, y3.gam), aes(x=range.gam, y=y3.gam)) +
  geom_line() +
  geom_vline(xintercept=mu.gam, size=0.2) +
  labs(title='', x='range', y='F(x)') +
  xlim(0, 15) +
  annotate('rect', xmin=mu.gam-sigma.gam, xmax=mu.gam+sigma.gam,
           ymin=-Inf, ymax=Inf, alpha=0.1) 

plot_grid(rplt.gam, dplt.gam, pplt.gam, nrow=3,
          labels=c('rgamma()', 'dgamma()', 'pgamma()'))
